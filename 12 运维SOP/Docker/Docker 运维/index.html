    Docker 运维

    1. Docker Image 管理¶

    1.1. Docker image 导出¶

    docker save -o mysql_8.0.36.tar.gz mysql:8.0.36

docker save -o zabbix-aent2_centos-6.4.10.tar.gz zabbix-agent2:centos-latest

docker save 951f02e66be2 -o xnet_monitor-agent2_centos-0.1.6.4.10.tar.gz xnet/monitor-agent2:centos-0.2.6.4.10

    1.1.1. 获取所有的镜像并导出¶

    IMAGES=$(docker images --format '{{.Repository}}:{{.Tag}}')

for IMAGE in $IMAGES; do  
  # 替换冒号为下划线，并将斜杠替换为下划线  FILENAME=${IMAGE//:/_}  
  FILENAME=${FILENAME//\//_}  
  docker save -o "/tmp/docker-images/${FILENAME}.tar.gz" $IMAGE  
done

    1.2. Docker image 导入¶

    docker load < mysql_8.0.36.tar.gz

    1.3. tag修改¶

    docker tag 411931b4d098 xnet/monitor-java-gateway:0.1.6.4.10

    1.4. image删除¶

    docker image rm xnet/zabbix-proxy-mysql:0.1.6.4.10

    2. 资源清理¶
    docker container stop $(docker container ls -a -q)
    docker container rm $(docker container ls -a -q)
    docker volume rm $(docker volume ls -q)
    docker network rm $(docker network ls -q)
    docker image rm $(docker image ls -a -q)
    停止所有容器运行 docker stop $(docker ps -a -q)
    删除所有停止运行的容器 docker rm $(docker ps -a -q)

    Docker 内 root

    docker exec -it --user $(username) $(containername) /bin/bash

docker exec -u 0 -it containername bash
