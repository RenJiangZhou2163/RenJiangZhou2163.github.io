    Alertmanager

    安装方式一¶

    export VERSION=0.26.0

curl -LO [https://github.com/prometheus/alertmanager/releases/download/v$VERSION/alertmanager-$VERSION.linux-amd64.tar.gz](https://github.com/prometheus/alertmanager/releases/download/v$VERSION/alertmanager-$VERSION.linux-amd64.tar.gz)

tar xvf alertmanager-$VERSION.linux-amd64.tar.gz

    安装方式二¶

    集群配置：¶

    vi /usr/lib/systemd/system/alertmanager.service

[Unit]
Description=Alertmanager handles alerts sent by client applications such as the Prometheus server.
Documentation=https://prometheus.io/docs/alerting/alertmanager/
After=network.target

[Service]
Type=simple
User=root
Group=root
ExecStart=/usr/local/alertmanager/alertmanager\
          --config.file=/etc/alertmanager/alertmanager.yml\
          --storage.path=/data/alertmanager/\
          --data.retention=120h\
          --cluster.listen-address=0.0.0.0:9094 \
          --cluster.peer=10.0.0.2:9094 \
          --cluster.peer=10.0.0.3:9094
Restart=on-failure

[Install]
WantedBy=multi-user.target
